By Object Literal

SYNTAX :

                "propertyOne":"valueOne",
                "propertyTwo":"valueTwo",
                "propertyThree":"valueThree"
                .
                .
                .
                .
                "propertyN":"valueN"
            };

            console.log(literalObjectName);
            console.log(typeof(literalObjectName));
